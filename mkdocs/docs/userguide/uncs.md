# Uncertainties

\begin{align}
\sigma(a+b) &= \sqrt{\sigma^2_a + \sigma^2_b}\\
\sigma(a-b) &= \sqrt{\sigma^2_a + \sigma^2_b}\\
\sigma(ab) &= \sqrt{a^2 \sigma_b^2 + b^2 \sigma_a^2}\\
\sigma(a/b) &= \sqrt{\frac{a^2 \sigma_b^2 + b^2 \sigma_a^2}{b^4}}\\
\sigma(a^b) &= \sqrt{a^{2b-2} (a^2 \sigma_b^2 (\ln{a})^2 + b^2 \sigma_a^2)}
\end{align}
